\section{Modellvereinfachung durch Betriebspunkt-Linearisierung}
\subsection{Betriebspunkt bzw. Arbeitspunkt eines Systems}
\begin{itemize}
 \item \underline{Betriebspunkt (BP)} = System-Ruhezustand, in dem die Ausgagsgröße y den stationären Sollwert $y_{s\infty}$ annimmt.
 \item \underline{Ruhezustand} (Ruhelage, Gleichgewichtslage, stationärer Zustand) = Systemzustand, in dem \underline{alle} Systemgrößen konstant
und damit ihre Zeitableitung gleich null sind. Zu seiner Bestimmung entweder die Systemgleichungen heranziehen und darin dei Ableitungen null 
setzen oder im Strukturbild in der Übertragunsfunktion $s=0$ setzen, außer beim $\boldsymbol{I-Glied}$:\\
 \item Dort muss die Eingangsgröße Null sein. Für die BP-Bestimmung dann noch $y \stackrel{!}{=} y_{s\infty}$ setzen.
\end{itemize}

\subsection{Durchführung der Linearisierung}
\begin{enumerate}
 \item \underline{Übergang von den Absolutgrößen $u(t)$, $y(t)$ zu den BP-Abweichungen}
	\[ \Delta u(t) = u(t) - u_B \\
	 \Delta y(t) = y(t) - y_B
	\]
Ein lineares ÜG $y(t) = \mathcal{S} \left \{ u(t) \right \} $ geht dabei über in 
\begin{align}
	\Delta y(t) &= y(t) - y_B \\
	&= \mathcal{S} \left \{ u(t)  \right \} - \mathcal{S} \left \{ u_B  \right \} \\
	&= \mathcal{S} \left \{ u_B  \right \} +  \mathcal{S} \left \{ \Delta u(t) \right \} - \mathcal{S} \left \{ \Delta u_B \right \}
\end{align}
\[ \rightarrow \Delta y(t) =  \mathcal{S} \left \{ \Delta u(t) \right \}\]
d.h. \underline{lineare ÜG bleiben unverändert} \\
Für ein nicht lineares ÜG folgt: \\
\begin{align}
	\Delta y(t) & = \mathcal{S} \left \{ u_B(t) + \Delta u(t) \right \} - \mathcal{S} \left \{ u_B \right \} \\
	& = \mathcal{\tilde S} \left \{ \Delta u(t) \right \} \not= \mathcal{\tilde S} \left \{ \Delta u(t) \right \}
\end{align}

d.h. \underline{Nichtlineare ÜG ändern dabei ihre Funktionalbezieung, bleiben aber i.A. nichtlinear} \\
